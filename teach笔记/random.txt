function randGenerator(n:any, sum:any) {
      let res = [];
      let sumTemp = sum;
      let resSum = 0;
      for (let i = 0; i < n - 1; i++) {
      let temp = Math.floor(Math.random() * (sumTemp / 2));
        res.push(temp);
        sumTemp -= temp;
        resSum += temp;
      }
      res.push(sum - resSum);
      return res;
    }
